## 1. 基础设置

初始化 Pygame、创建窗口、控制游戏循环。

```python
import pygame
import sys

pygame.init()
screen = pygame.display.set_mode((800, 600))
pygame.display.set_caption("我的游戏")
clock = pygame.time.Clock()

running = True
while running:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    screen.fill((0, 0, 0))  # 清屏
    pygame.display.flip()
    clock.tick(60)  # 控制帧率

pygame.quit()
sys.exit()
```

## 6. 碰撞检测

使用矩形或圆形检测碰撞。

```python
rect1 = pygame.Rect(50, 50, 50, 50)
rect2 = pygame.Rect(70, 60, 50, 50)

if rect1.colliderect(rect2):
    print("矩形碰撞！")

# 圆形碰撞
def circle_collision(c1, c2):
    distance = ((c1[0]-c2[0])**2 + (c1[1]-c2[1])**2)**0.5
    return distance < (c1[2] + c2[2])
```

## 8. 精灵与精灵组

使用 Sprite 类管理游戏对象。

```python
class Player(pygame.sprite.Sprite):
    def __init__(self):
        super().__init__()
        self.image = pygame.Surface((50, 50))
        self.image.fill((0, 255, 0))
        self.rect = self.image.get_rect()
        self.rect.center = (400, 300)

    def update(self):
        keys = pygame.key.get_pressed()
        if keys[pygame.K_LEFT]:
            self.rect.x -= 5
        if keys[pygame.K_RIGHT]:
            self.rect.x += 5

all_sprites = pygame.sprite.Group()
player = Player()
all_sprites.add(player)

# 更新和绘制
all_sprites.update()
all_sprites.draw(screen)
```

## 完整示例：简易躲避游戏

```python
import pygame, random, sys

pygame.init()
screen = pygame.display.set_mode((800, 600))
clock = pygame.time.Clock()
font = pygame.font.Font(None, 36)

player = pygame.Rect(375, 500, 50, 50)
enemies = []
score = 0
enemy_timer = 0
running = True

while running:
    clock.tick(60)
    screen.fill((0, 0, 0))

    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    keys = pygame.key.get_pressed()
    if keys[pygame.K_LEFT] and player.left > 0:
        player.x -= 5
    if keys[pygame.K_RIGHT] and player.right < 800:
        player.x += 5

    enemy_timer += 1
    if enemy_timer > 60:
        enemies.append(pygame.Rect(random.randint(0, 750), -50, 50, 50))
        enemy_timer = 0

    for enemy in enemies[:]:
        enemy.y += 5
        pygame.draw.rect(screen, (255, 0, 0), enemy)
        if player.colliderect(enemy):
            running = False
        if enemy.top > 600:
            enemies.remove(enemy)
            score += 1

    pygame.draw.rect(screen, (0, 255, 0), player)
    score_text = font.render(f"得分: {score}", True, (255, 255, 255))
    screen.blit(score_text, (10, 10))

    pygame.display.flip()

pygame.quit()
sys.exit()
```
